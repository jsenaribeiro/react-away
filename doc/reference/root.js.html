import { jsx as _jsx } from "react/jsx-runtime";
import React from 'react';
import ReactClassicDOM from 'react-dom/client';
import { Provider } from './storer/provider';
import { I18nFluent } from './storer/i18n';
import formerProper from './binder';
import stylerProper from './styler';
export const render = {
    context: {},
    stricted: false,
    directives: [
        formerProper,
        stylerProper
    ]
};
/**
 * React Away startup class
 */
export default class ReactDOM {
    static createRoot(strict, query, app, Store, directives = []) {
        if (!!directives.length)
            directives.forEach((d) => render.directives.push(d));
        render.context = new Store();
        render.stricted = strict;
        const root = document.querySelector(query);
        const wrap = _jsx(Provider, { src: render.context, children: app });
        const main = strict ? _jsx(React.StrictMode, { children: wrap }) : wrap;
        ReactClassicDOM.createRoot(root).render(main);
        return new I18nFluent();
    }
}
