/** @module Storing */
import React from "react";
import { cloneComponent } from '../cloner';
import { createProxy } from './proxy';
import { context } from "../shared";
import { listener } from './listener';
import { router } from "../router";
/**
 * React Away provider class
 */
export class Provider extends React.Component {
    constructor(props) {
        super(props);
        const refresh = () => this.setState({ tick: new Date().getTime() });
        window.store = createProxy(props.src);
        context.render = (timeout = 1) => !context.enable ? undefined
            : setTimeout(() => refresh(), timeout);
        const route = router.getRoute();
        const value = router.getRoute("*");
        listener.dispatch(route, value);
    }
    render() {
        listener.dispatch("refresh", {});
        return cloneComponent(this);
    }
    componentDidMount() { listener.register(true); listener.dispatch("mounted"); }
    componentWillUnmount() { listener.register(false); listener.dispatch("unmount"); }
    componentDidCatch = (error, info) => listener.dispatch("catched", { error, errorInfo: info });
    componentDidUpdate = (props, state, snapshot) => listener.dispatch("updated", { prevProps: props, prevState: state, snapshot });
}
