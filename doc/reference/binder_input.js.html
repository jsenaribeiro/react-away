/** @module Binding */
import { valuer } from "../storer/valuer";
const isTwoWayDataBindingElement = (args) => ["input", "select", "textarea"].includes(args.tag);
/**
 * Directive for input two-way data binding
 * @param {object} props current props
 * @param {DirectiveArgs} args directive args
 * @returns handles bind props for two-way data binding
 */
export function twoWayDataBindingControl(props, args) {
    if (!props?.bind)
        return props;
    if (!isTwoWayDataBindingElement(args))
        return props;
    const field = props.bind;
    const state = props.local ? window['state'][args.id] : window.store;
    const value = valuer(state, field);
    const event = (e) => valuer(state, field, e.target.value);
    const model = { props, value, event };
    return args.tag == "input" && props.type == "radio" ? radio(model)
        : args.tag == "input" && props.type == "checkbox" ? checkbox(model)
            : args.tag == "input" ? input(model)
                : args.tag == "select" ? select(model)
                    : args.tag == "textarea" ? textarea(model)
                        : props;
}
const input = (h) => ({ ...h.props, value: h.value, onInput: h.event });
const textarea = (h) => ({ ...h.props, value: h.value, onChange: h.event });
const select = (h) => ({ ...h.props, value: h.value, onSelect: h.event });
const radio = (h) => ({ ...h.props, checked: h.value, onChange: h.event });
const checkbox = (h) => ({ ...h.props, checked: h.value, onInput: h.event });
